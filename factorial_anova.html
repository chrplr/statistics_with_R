Factorial ANOVAs

rm(list = ls())
require(ez)

## Loading required package: ez

Generation of a dataset S

subject = factor(paste("sub", 1:30, sep = ""))
A = gl(3, 10, 30, labels = c("a1", "a2", "a3"))
B = gl(2, 5, 30, labels = c("a2", "b2"))
x = rnorm(30, mean = 10) + 1 * (A == "a1" & B == "b2")
dat = data.frame(subject, A, B, x)

rm(subject, A, B, x)
attach(dat)
